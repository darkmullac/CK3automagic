namespace = auto_build
auto_build.0003 = {
	type = character_event
	trigger = {
		OR = {
			has_character_flag = enable_auto_build
			has_character_flag = enable_vassal_auto_build
		}
		
	}
	hidden = yes
	immediate = {
		hidden_effect = {
			trigger_event = {
				id = auto_build.0004
				days = 15

}

}

# The vanilla build order seemed odd, with cattle pastures first.
# Event IDs may appear out of order while priorities are adjusted.
# You can freely reorder them by copy-pasting.
# Tip: Indian wind furnaces lack innovation checks; build level one manually.
# I advise you to use Visual Studio Code to edit.

auto_build.0001 = {
	type = character_event
	title = auto_build.0001.t
	desc = auto_build.0001.desc
	weight_multiplier = {
		base = 1
	}
	override_background = {
		reference = terrain
	}
	left_portrait = {
		character = scope:root
	}
	immediate = {
		hidden_effect = {
			trigger_event = {
				id = auto_build.0003
				days = 5
			}
		}
	}
	option = {
		name = auto_build.0001.a
		add_character_flag = enable_auto_build
	}
	option = {
		name = auto_build.0001.b
		add_character_flag = enable_vassal_auto_build
	}
}
